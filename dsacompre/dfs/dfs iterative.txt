// C++ program to print bfs traversal from a given vertex in a  given graph
#include<bits/stdc++.h>

using namespace std;

vector <int> g[10000000];
bool visited[10000000];

void dfs(int v)
{

   stack<int> q;
   q.push(v);
   
   visited[v]=true;
   while( !q.empty())
    {
    int temp =q.top();
    q.pop();
    visited[temp]=true;
    cout<<temp<<" ";
    for(int i=0;i<g[temp].size();i++)
    {
        if( ! visited[g[temp][i]])
            q.push(g[temp][i]);
            visited[g[temp][i]]=true;
            
    }
}

}

int main()
{
int n,m;
cin>>n>>m;
int i;
int l,r;
for(i=0;i<=n;i++){
    visited[i]=false;
}
for(i=1;i<=m;i++)
{
  cin>>l>>r;
  g[l].push_back(r);
 // g[r].push_back(l);
}
dfs(1);

}
